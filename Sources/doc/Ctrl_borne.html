def ask_adhesion n
        case n
                when 1
                        quest=
                        "Par la suite nous allons receuillir vos informations d'abonne.\nVeuillez specifier votre nom : "
                        vue_borne.insert quest
                when 2
                        quest=
                        "Veuillez specifier votre prenom : "
                        vue_borne.insert quest
                when 3
                        quest=
                        "Veuillez specifier votre adresse : "
                        vue_borne.insert quest
                when 4
        end
end
