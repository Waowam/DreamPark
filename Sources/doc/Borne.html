def apply_scenarios
        case get_num_scenario
                when 1
                        controleur.ask_abonnement
                when 2
                        controleur.ask_services
                when 3
                        controleur.ask_upgrade
                when 4
                        controleur.ask_paiement
                when 5
                        controleur.ask_adhesion 1
                when (-1)
                        controleur.say_bybye
                        controleur.reset_view
                        @autorisation=true
                        @acces.est_entre
        end
end
